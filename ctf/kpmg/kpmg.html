def xor_decrypt(ciphertext, key):
    decrypted = ""
    for i in range(0, len(ciphertext), 2):
        hex_pair = ciphertext[i:i+2]
        decrypted += chr(int(hex_pair, 16) ^ ord(key))
    return decrypted
